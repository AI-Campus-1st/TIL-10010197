import random

# 원본은 그대로 두고 정리한 결과를 돌려줌 
sorted_nums = sorted(presenter)

#sort 경우
import random


nums = list(range(1, 24))
presenter = random.sample(nums, 4)

sorted_nums = presenter.sort() #정렬만 하고 따로 반환하는 건 없음
print(presenter) 
print(sorted_nums) # None


sorted 함수는 기존의 리스트를 유지한 채, 새로 정렬된 리스트를 반환한다. 
sort 함수는 기존의 리스트를 정렬하고, 반환하는 리스트는 없다.
